23void ProxyServer::setIntercept(bool on)
24{
25 m_intercept = on;
26 for (auto *conn : m_connections)
27 conn->setIntercept(on);
28
29 if (!on) {
30 // Release the currently active intercepted connection (if any)
31 if (m_activeId >= 0) {
32 auto *conn = m_connections.value(m_activeId, nullptr);
33 if (conn) conn->releaseIfHeld();
34 m_activeId = -1;
35 }
36 // Release every connection waiting in the queue
37 while (!m_interceptQueue.isEmpty()) {
38 auto item = m_interceptQueue.dequeue();
39 auto *conn = m_connections.value(item.connId, nullptr);
40 if (conn) conn->releaseIfHeld();
41 }
42 emit queueChanged(0);
43 emit nextIntercepted(-1, {});
44 }
45}

47void ProxyServer::incomingConnection(qintptr handle)
48{
49 auto *conn = new ProxyConnection(handle, m_intercept, this);
50
51 connect(conn, &ProxyConnection::intercepted, this,
52 [this](int id, QByteArray raw) {
53 if (m_activeId < 0) {
54 // No request currently shown — show this one immediately
55 m_activeId = id;
56 emit requestIntercepted(id, raw);
57 } else {
58 // Another request is being reviewed — queue it
59 m_interceptQueue.enqueue({id, raw});
60 // nextIntercepted with a queue-depth hint so UI can show counter
61 emit nextIntercepted(-2, raw); // -2 = "queued, not yet shown"
62 }
63 });
64
65 connect(conn, &ProxyConnection::finished, this,
66 [this](TrafficRecord rec) {
67 emit requestFinished(rec);
68 });
69
70 connect(conn, &ProxyConnection::done, this,
71 [this](int id) {
72 m_connections.remove(id);
73 if (m_activeId == id) {
74 m_activeId = -1;
75 promoteNext();
76 }
77 // Also scrub from queue if the connection died on its own
78 QQueue<PendingItem> cleaned;
79 for (auto &item : m_interceptQueue)
80 if (item.connId != id) cleaned.enqueue(item);
81 m_interceptQueue = cleaned;
82 });
83
84 m_connections.insert(conn->id(), conn);
85}

88void ProxyServer::promoteNext()
89{
90 if (m_interceptQueue.isEmpty()) {
91 emit nextIntercepted(-1, {}); // -1 = queue empty
92 return;
93 }
94 auto item = m_interceptQueue.dequeue();
95 m_activeId = item.connId;
96 emit requestIntercepted(item.connId, item.raw);
97}

99void ProxyServer::forwardPendingRequest(const QByteArray &rawRequest)
100{
101 if (m_activeId < 0) return;
102 auto *conn = m_connections.value(m_activeId, nullptr);
103
104 // Clear active slot BEFORE calling promoteNext so it can assign the next id
105 m_activeId = -1;
106
107 if (conn) conn->forwardRequest(rawRequest);
108
109 promoteNext();
110
111 emit queueChanged(m_interceptQueue.size() + (m_activeId >= 0 ? 1 : 0));
112}
