23void RepeaterClient::send(const QByteArray &rawRequest,
24 const QString &host, int port, bool useSsl)
25{
26 hardReset(); // kill any previous socket / timers cleanly
27
28 m_request = rawRequest;
29 m_response.clear();
30 m_host = host;
31 m_port = port;
32 m_useSsl = useSsl;
33 m_finished = false;
34
35 if (useSsl) {
36 auto *ssl = new QSslSocket(this);
37 ssl->setPeerVerifyMode(QSslSocket::VerifyNone);
38 m_socket = ssl;
39 connect(ssl, &QSslSocket::encrypted, this, &RepeaterClient::onSslConnected);
40 } else {
41 m_socket = new QTcpSocket(this);
42 connect(m_socket, &QTcpSocket::connected, this, &RepeaterClient::onConnected);
43 }
44
45 connect(m_socket, &QTcpSocket::readyRead, this, &RepeaterClient::onReadyRead);
46 connect(m_socket, &QTcpSocket::disconnected, this, &RepeaterClient::onDisconnected);
47 connect(m_socket,
48 QOverload<QAbstractSocket::SocketError>::of(&QTcpSocket::errorOccurred),
49 this, &RepeaterClient::onSocketError);
50
51 m_overallTimer->start(20000);
52
53 if (useSsl)
54 qobject_cast<QSslSocket*>(m_socket)->connectToHostEncrypted(host, static_cast<quint16>(port));
55 else
56 m_socket->connectToHost(host, static_cast<quint16>(port));
57}

63void RepeaterClient::doSend()
64{
65 QByteArray req = m_request;
66
67 // 1. Нормализация переносов строк (ОЧЕНЬ ВАЖНО для HTTP)
68 // QPlainTextEdit отдает \n, а стандарт HTTP требует \r\n
69 req.replace("\r\n", "\n"); // Убираем старые \r\n, чтобы не задвоить
70 req.replace("\n", "\r\n"); // Делаем все переносы по стандарту
71
72 // Убеждаемся, что пустая строка (окончание заголовков) вообще есть
73 if (!req.contains("\r\n\r\n")) {
74 if (!req.endsWith("\r\n")) req.append("\r\n");
75 req.append("\r\n");
76 }
77
78 // 2. Разделяем запрос на заголовки и тело для безопасной модификации
79 int blankIdx = req.indexOf("\r\n\r\n");
80 QByteArray hdrs = req.left(blankIdx);
81 QByteArray body = req.mid(blankIdx + 4);
82
83 QByteArray lowerHdrs = hdrs.toLower();
84
85 // 3. Форсируем Connection: close
86 int connIdx = lowerHdrs.indexOf("\nconnection:");
87 if (connIdx >= 0) {
88 connIdx++; // сдвигаемся на начало слова 'c'
89 int lineEnd = hdrs.indexOf("\r\n", connIdx);
90 if (lineEnd >= 0) {
91 hdrs.replace(connIdx, lineEnd - connIdx, "Connection: close");
92 }
93 } else {
94 hdrs.append("\r\nConnection: close");
95 }
96
97 // 4. Авто-пересчет Content-Length
98 // Если ты стер пару символов в теле через UI, сервер больше не будет висеть в ожидании
99 lowerHdrs = hdrs.toLower(); // Обновляем после возможной вставки Connection
100 int clIdx = lowerHdrs.indexOf("\ncontent-length:");
101 if (clIdx >= 0) {
102 clIdx++;
103 int lineEnd = hdrs.indexOf("\r\n", clIdx);
104 if (lineEnd >= 0) {
105 hdrs.replace(clIdx, lineEnd - clIdx, "Content-Length: " + QByteArray::number(body.size()));
106 }
107 }
108
109 // Собираем идеальный HTTP-запрос обратно
110 req = hdrs + "\r\n\r\n" + body;
111
112 // Отправляем и проталкиваем буфер
113 m_socket->write(req);
114 m_socket->flush();
115}
116void RepeaterClient::finish()
117{
118 if (m_finished) return;
119 m_finished = true;
120
121 m_overallTimer->stop();
122
123 // Drain any remaining bytes before killing the socket
124 if (m_socket)
125 m_response += m_socket->readAll();
126
127 QByteArray resp = m_response;
128
129 // Destroy socket cleanly without triggering more signals
130 if (m_socket) {
131 m_socket->disconnect(this);
132 m_socket->abort();
133 m_socket->deleteLater();
134 m_socket = nullptr;
135 }
136
137 emit responseReceived(resp);
138}

160void RepeaterClient::onReadyRead()
161{
162 if (!m_socket) return;
163 m_response += m_socket->readAll();
164
165 // Try to detect end of response via Content-Length
166 int headerEnd = m_response.indexOf("\r\n\r\n");
167 if (headerEnd < 0) return;
168
169 QByteArray hdrs = m_response.left(headerEnd).toLower();
170
171 // Chunked: wait for terminal chunk
172 if (hdrs.contains("transfer-encoding: chunked")) {
173 // Улучшенная проверка конца чанка
174 if (m_response.endsWith("0\r\n\r\n"))
175 finish();
176 return;
177 }
178
179 // Content-Length
180 int clPos = hdrs.indexOf("content-length:");
181 if (clPos >= 0) {
182 int eol = hdrs.indexOf('\n', clPos);
183 bool ok = false;
184 int cl = hdrs.mid(clPos + 15, eol - clPos - 15).trimmed().toInt(&ok);
185 if (ok) {
186 int bodyReceived = m_response.size() - (headerEnd + 4);
187 if (bodyReceived >= cl)
188 finish();
189 }
190 return;
191 }
192
193 // Unknown length — rely on disconnect
194}

202void RepeaterClient::onSocketError(QAbstractSocket::SocketError err)
203{
204 // RemoteHostClosedError fires before disconnected on many platforms
205 if (err == QAbstractSocket::RemoteHostClosedError) {
206 finish();
207 return;
208 }
209
210 if (m_finished) return;
211 m_finished = true;
212 m_overallTimer->stop();
213
214 QString msg = m_socket ? m_socket->errorString() : "socket error";
215 if (m_socket) {
216 m_socket->disconnect(this);
217 m_socket->abort();
218 m_socket->deleteLater();
219 m_socket = nullptr;
220 }
221 emit errorOccurred(msg);
222}

224void RepeaterClient::onOverallTimeout()
225{
226 if (m_finished) return;
227 if (!m_response.isEmpty())
228 finish(); // return partial response
229 else {
230 m_finished = true;
231 hardReset();
232 emit errorOccurred("Timed out — no response in 20 s");
233 }
