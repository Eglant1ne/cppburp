32// ─────────────────────────────────────────────────────────────────────────────
33// Client → Proxy (incoming data from browser)
34// ─────────────────────────────────────────────────────────────────────────────
35void ProxyConnection::onClientData()
36{
37 m_clientBuffer += m_clientSocket->readAll();
38
39 // In tunnel mode just forward bytes to target
40 if (m_tunnelMode) {
41 if (m_targetSocket && m_targetSocket->isOpen())
42 m_targetSocket->write(m_clientBuffer);
43 m_clientBuffer.clear();
44 return;
45 }
46
47 // Try to parse HTTP header
48 if (!m_requestParsed) {
49 try {
50 if (!HttpParser::parse(m_clientBuffer, m_parsedRequest))
51 return; // need more data
52 } catch (const std::invalid_argument& ex) {
53 qWarning("ProxyConnection: bad request — %s", ex.what());
54 dropConnection();
55 return;
56 }
57 m_requestParsed = true;
58 }
59
60 // CONNECT → set up TLS tunnel
61 if (m_parsedRequest.isConnect) {
62 if (m_clientSocket->isOpen())
63 m_clientSocket->write("HTTP/1.1 200 Connection established\r\n\r\n");
64 m_clientBuffer.clear();
65 connectToTarget(m_parsedRequest.host, m_parsedRequest.port, /*tls=*/true);
66 return;
67 }
68
69 // Normal HTTP request
70 if (m_interceptOn) {
71 emit intercepted(m_id, m_clientBuffer);
72 return;
73 }
74
75 connectToTarget(m_parsedRequest.host, m_parsedRequest.port, m_parsedRequest.isTls);
76}

81void ProxyConnection::connectToTarget(const QString &host, int port, bool tls)
82{
83 if (tls) {
84 auto *ssl = new QSslSocket(this);
85 ssl->setPeerVerifyMode(QSslSocket::VerifyNone);
86 m_targetSocket = ssl;
87 connect(ssl, &QSslSocket::encrypted,
88 this, &ProxyConnection::onSslEstablished);
89 } else {
90 m_targetSocket = new QTcpSocket(this);
91 }
92
93 connect(m_targetSocket, &QTcpSocket::connected,
94 this, &ProxyConnection::onTargetConnected);
95 connect(m_targetSocket, &QTcpSocket::readyRead,
96 this, &ProxyConnection::onTargetData);
97 connect(m_targetSocket, &QTcpSocket::disconnected,
98 this, &ProxyConnection::onTargetDisconnected);
99
100 if (tls) {
101 auto *ssl = qobject_cast<QSslSocket*>(m_targetSocket);
102 ssl->connectToHostEncrypted(host, static_cast<quint16>(port));
103 } else {
104 m_targetSocket->connectToHost(host, static_cast<quint16>(port));
105 }
106}

148void ProxyConnection::onTargetDisconnected()
149{
150 if (!m_tunnelMode && m_requestParsed) {
151 TrafficRecord rec;
152 rec.id = m_id;
153 rec.time = m_startTime;
154 rec.method = m_parsedRequest.method;
155 rec.host = m_parsedRequest.host;
156 rec.size = m_responseBuffer.size();
157
158 QString fullUrl = m_parsedRequest.url;
159 if (fullUrl.startsWith("http://") || fullUrl.startsWith("https://")) {
160 int slashPos = fullUrl.indexOf('/', 8);
161 rec.path = slashPos >= 0 ? fullUrl.mid(slashPos) : "/";
162 } else {
163 rec.path = fullUrl;
164 }
165
166 rec.rawRequest = QString::fromLatin1(m_parsedRequest.raw);
167 rec.rawResponse = QString::fromLatin1(m_responseBuffer);
168
169 int statusStart = m_responseBuffer.indexOf("HTTP/");
170 if (statusStart >= 0) {
171 int sp1 = m_responseBuffer.indexOf(' ', statusStart);
172 if (sp1 >= 0) {
173 int sp2 = m_responseBuffer.indexOf(' ', sp1 + 1);
174 rec.status = m_responseBuffer.mid(sp1 + 1, sp2 - sp1 - 1).toInt();
175 }
176 }
177
178 emit finished(rec);
179 }
180
181 if (m_clientSocket && m_clientSocket->isOpen())
182 m_clientSocket->disconnectFromHost();
183
184 emit done(m_id);
185 deleteLater();
186}

196// ─────────────────────────────────────────────────────────────────────────────
197// Intercept actions (called from UI)
198// ─────────────────────────────────────────────────────────────────────────────
199
200// Normalise raw bytes coming from QPlainTextEdit before sending to origin server.
201// Fixes:
202// 1. \n → \r\n (QPlainTextEdit strips \r)
203// 2. Absolute URL → relative path (browsers send "GET http://host/path" to
204// proxies, but origin servers only accept "GET /path")
205// 3. Connection: close (prevents keep-alive stalls)
206// 4. Content-Length recalculated after body edits
207static QByteArray normalizeRequest(const QByteArray &raw)
208{
209 QByteArray req = raw;
210
211 // 1. Normalize line endings
212 req.replace("\r\n", "\n");
213 req.replace("\n", "\r\n");
214
215 // Ensure header/body separator
216 if (!req.contains("\r\n\r\n")) {
217 if (!req.endsWith("\r\n")) req.append("\r\n");
218 req.append("\r\n");
219 }
220
221 int blankIdx = req.indexOf("\r\n\r\n");
222 QByteArray hdrs = req.left(blankIdx);
223 QByteArray body = req.mid(blankIdx + 4);
224
225 // 2. Strip absolute URL in the request line → relative path
226 // "GET http://example.com/path?q=1 HTTP/1.1"
227 // becomes "GET /path?q=1 HTTP/1.1"
228 {
229 int sp1 = hdrs.indexOf(' '); // after method
230 int sp2 = hdrs.indexOf(' ', sp1 + 1); // after URL
231 if (sp1 > 0 && sp2 > sp1) {
232 QByteArray url = hdrs.mid(sp1 + 1, sp2 - sp1 - 1);
233 if (url.startsWith("http://") || url.startsWith("https://")) {
234 int schemeEnd = url.indexOf("://") + 3;
235 int pathStart = url.indexOf('/', schemeEnd);
236 QByteArray rel = (pathStart >= 0) ? url.mid(pathStart) : QByteArray("/");
237 hdrs.replace(sp1 + 1, url.size(), rel);
238 }
239 }
240 }
241
242 QByteArray lowerHdrs = hdrs.toLower();
243
244 // 3. Force Connection: close
245 int connIdx = lowerHdrs.indexOf("\nconnection:");
246 if (connIdx >= 0) {
247 connIdx++;
248 int lineEnd = hdrs.indexOf("\r\n", connIdx);
249 if (lineEnd >= 0)
250 hdrs.replace(connIdx, lineEnd - connIdx, "Connection: close");
251 } else {
252 hdrs.append("\r\nConnection: close");
253 }
254
255 // 4. Recalculate Content-Length
256 lowerHdrs = hdrs.toLower();
257 int clIdx = lowerHdrs.indexOf("\ncontent-length:");
258 if (clIdx >= 0) {
259 clIdx++;
260 int lineEnd = hdrs.indexOf("\r\n", clIdx);
261 if (lineEnd >= 0)
262 hdrs.replace(clIdx, lineEnd - clIdx,
263 "Content-Length: " + QByteArray::number(body.size()));
264 }
265
266 return hdrs + "\r\n\r\n" + body;
267}

269void ProxyConnection::forwardRequest(const QByteArray &modifiedRequest)
270{
271 QByteArray req = normalizeRequest(modifiedRequest);
272
273 m_clientBuffer = req;
274
275 // Re-parse to get up-to-date host/port from the edited request.
276 HttpRequest parsed;
277 try {
278 if (HttpParser::parse(req, parsed))
279 m_parsedRequest = parsed;
280 } catch (const std::invalid_argument& ex) {
281 qWarning("ProxyConnection::forwardRequest — bad request: %s", ex.what());
282 dropConnection();
283 return;
284 }
285
286 // Destroy any previous target socket so we don't leak/double-connect
287 if (m_targetSocket) {
288 m_targetSocket->disconnect();
289 m_targetSocket->deleteLater();
290 m_targetSocket = nullptr;
291 }
292
293 m_responseBuffer.clear();
294 connectToTarget(m_parsedRequest.host, m_parsedRequest.port, m_parsedRequest.isTls);
295}

297void ProxyConnection::releaseIfHeld()
298{
299 // A connection is "held" when intercept fired but the user has not yet
300 // forwarded or dropped it. m_clientBuffer still holds the original raw
301 // request at that point and no target socket has been created yet.
302 if (m_targetSocket || m_tunnelMode || !m_requestParsed)
303 return; // already connecting/connected, or nothing parsed yet
304
305 // Forward using the bytes we already have
306 connectToTarget(m_parsedRequest.host, m_parsedRequest.port, m_parsedRequest.isTls);
307}
