7bool HttpParser::parse(const QByteArray& data, HttpRequest& out)
8{
9 out.raw = data;
10
11 // Find end of headers (\r\n\r\n)
12 int headerEnd = data.indexOf("\r\n\r\n");
13 if (headerEnd < 0) return false;
14
15 QByteArray headerPart = data.left(headerEnd);
16 out.body = data.mid(headerEnd + 4);
17
18 QList<QByteArray> lines = headerPart.split('\n');
19 if (lines.isEmpty()) return false;
20
21 // Request line
22 QString requestLine = QString::fromLatin1(lines[0]).trimmed();
23 QStringList parts = requestLine.split(' ');
24 if (parts.size() < 3)
25 throw std::invalid_argument(
26 "Malformed HTTP request line: fewer than three tokens");
27
28 out.method = parts[0].toUpper();
29 out.url = parts[1];
30 out.version = parts[2];
31 out.isConnect = (out.method == "CONNECT");
32
33 // Headers
34 for (int i = 1; i < lines.size(); ++i) {
35 QString line = QString::fromLatin1(lines[i]).trimmed();
36 int colonIdx = line.indexOf(':');
37 if (colonIdx < 0) continue;
38 QString key = line.left(colonIdx).trimmed().toLower();
39 QString value = line.mid(colonIdx + 1).trimmed();
40 out.headers[key] = value;
41 }
42
43 // Resolve host/port
44 QString hostHeader = out.headers.value("host");
45 if (out.isConnect) {
46 // CONNECT host:port → TLS tunnel
47 splitHostPort(out.url, out.host, out.port, /*isTls=*/true);
48 out.isTls = true;
49 } else if (!hostHeader.isEmpty()) {
50 bool defaultTls = out.url.startsWith("https://");
51 splitHostPort(hostHeader, out.host, out.port, defaultTls);
52 out.isTls = defaultTls;
53 }
54
55 return true;
56}

58void HttpParser::splitHostPort(const QString& hostPort, QString& host, int& port, bool isTls)
59{
60 // Handle IPv6 [::1]:8080
61 if (hostPort.startsWith('[')) {
62 int closeBracket = hostPort.indexOf(']');
63 if (closeBracket < 0) {
64 host = hostPort;
65 port = isTls ? 443 : 80;
66 return;
67 }
68 host = hostPort.mid(1, closeBracket - 1);
69 QString rest = hostPort.mid(closeBracket + 1);
70 if (rest.startsWith(':'))
71 port = rest.mid(1).toInt();
72 else
73 port = isTls ? 443 : 80;
74 return;
75 }
76
77 int colonIdx = hostPort.lastIndexOf(':');
78 if (colonIdx > 0) {
79 bool ok = false;
80 int p = hostPort.mid(colonIdx + 1).toInt(&ok);
81 if (ok) {
82 host = hostPort.left(colonIdx);
83 port = p;
84 return;
85 }
86 }
87 host = hostPort;
88 port = isTls ? 443 : 80;
89}
