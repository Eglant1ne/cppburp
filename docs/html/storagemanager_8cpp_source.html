17bool StorageManager::open(const QString& filePath)
18{
19 m_db = QSqlDatabase::addDatabase("QSQLITE", "proxylab_conn");
20 m_db.setDatabaseName(filePath);
21 if (!m_db.open())
22 throw std::runtime_error(
23 QString("StorageManager: cannot open database: %1")
24 .arg(m_db.lastError().text())

30void StorageManager::createSchema()
31{
32 QSqlQuery q(m_db);
33 q.exec("PRAGMA journal_mode=WAL;");
34 q.exec(R"(
35 CREATE TABLE IF NOT EXISTS traffic (
36 id INTEGER PRIMARY KEY AUTOINCREMENT,
37 time TEXT,
38 method TEXT,
39 host TEXT,
40 path TEXT,
41 status INTEGER,
42 size INTEGER,
43 raw_request TEXT,
44 raw_response TEXT
45 )
46 )");
47 q.exec(R"(
48 CREATE TABLE IF NOT EXISTS groups (
49 id INTEGER PRIMARY KEY AUTOINCREMENT,
50 name TEXT NOT NULL
51 )
52 )");
53 q.exec(R"(
54 CREATE TABLE IF NOT EXISTS saved_requests (
55 id INTEGER PRIMARY KEY AUTOINCREMENT,
56 group_id INTEGER REFERENCES groups(id) ON DELETE CASCADE,
57 name TEXT,
58 raw_request TEXT,
59 host TEXT,
60 port INTEGER,
61 use_ssl INTEGER
62 )
63 )");
64}

68void StorageManager::saveRecord(const TrafficRecord& rec)
69{
70 QSqlQuery q(m_db);
71 q.prepare(R"(
72 INSERT INTO traffic (time,method,host,path,status,size,raw_request,raw_response)
73 VALUES (:time,:method,:host,:path,:status,:size,:req,:resp)
74 )");
75 q.bindValue(":time", rec.time.toString(Qt::ISODate));
76 q.bindValue(":method", rec.method);
77 q.bindValue(":host", rec.host);
78 q.bindValue(":path", rec.path);
79 q.bindValue(":status", rec.status);
80 q.bindValue(":size", rec.size);
81 q.bindValue(":req", rec.rawRequest);
82 q.bindValue(":resp", rec.rawResponse);
83 if (!q.exec())
84 qWarning() << "saveRecord:" << q.lastError().text();
85}

87QVector<TrafficRecord> StorageManager::loadHistory(int limit)
88{
89 QVector<TrafficRecord> result;
90 QSqlQuery q(m_db);
91 q.prepare("SELECT id,time,method,host,path,status,size,raw_request,raw_response "
92 "FROM traffic ORDER BY id DESC LIMIT :lim");
93 q.bindValue(":lim", limit);
94 q.exec();
95 while (q.next()) {
96 TrafficRecord r;
97 r.id = q.value(0).toInt();
98 r.time = QDateTime::fromString(q.value(1).toString(), Qt::ISODate);
99 r.method = q.value(2).toString();
100 r.host = q.value(3).toString();
101 r.path = q.value(4).toString();
102 r.status = q.value(5).toInt();
103 r.size = q.value(6).toInt();
104 r.rawRequest = q.value(7).toString();
105 r.rawResponse = q.value(8).toString();
106 result.append(r);
107 }
108 return result;
109}
